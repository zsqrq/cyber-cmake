1262  template<class _T>
1263  FastCdr& deserializeSequence(_T *&sequence_t, size_t &numElements)
1264  {
1265  uint32_t seqLength = 0;
1266  state state(*this);
1267 
1268  deserialize(seqLength);
1269 
1270  try
1271  {
1272  sequence_t = (_T*)calloc(seqLength, sizeof(_T));
1273  deserializeArray(sequence_t, seqLength);
1274  }
1275  catch(eprosima::fastcdr::exception::Exception &ex)
1276  {
1277  free(sequence_t);
1278  sequence_t = NULL;
1279  setState(state);
1280  ex.raise();
1281  }
1282 
1283  numElements = seqLength;
1284  return *this;
1285  }
